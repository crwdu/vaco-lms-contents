class LinkedList {
  add(value) {
    let newNode = new Node(value);

    if (!this.head) {
      this.head = newNode;
      return this.head;
    }

    let tail = this.head;

    while (tail.next !== null) {
      tail = tail.next;
    }

    tail.next = newNode;

    return this.head;
  }

  size() {
    let count = 0;
    let node = this.head;

    while (node) {
      count++;
      node = node.next;
    }

    return count;
  }
}
